b) **Add Product + Offer JSON-LD:**
Add a helper function in the frontmatter:
```typescript
function getAvailability(stock: number, madeToOrder: boolean): string {
  if (stock > 0) return "https://schema.org/InStock";
  if (madeToOrder) return "https://schema.org/MadeToOrder";
  return "https://schema.org/SoldOut";
}

const productSchema = {
  "@context": "https://schema.org",
  "@type": "Product",
  "name": name,
  "image": `${siteUrl}${images[0]}`,
  "description": description || `Handcrafted ${material.toLowerCase()} piece from the ${collection} collection.`,
  "sku": code,
  "brand": {
    "@type": "Brand",
    "name": "Redcocoon Pottery"
  },
  "material": material,
  "offers": {
    "@type": "Offer",
    "price": price,
    "priceCurrency": currency,
    "availability": getAvailability(stock, isMTO),
    "url": `${siteUrl}/shop/${product.id.replace(/\.md$/, '')}`,
    "seller": {
      "@type": "Organization",
      "name": "Redcocoon Pottery"
    }
  }
};
```

Inject the schema inside the BaseLayout, right after the opening `<BaseLayout ...>` tag but before the container div, using a `<Fragment slot="head">` approach. However, since BaseLayout does not have a head slot, inject it as a `<script>` tag inside the body (this is valid and Google reads it fine):

```html
<script type="application/ld+json" set:html={JSON.stringify(productSchema)} />
```

Place this line right after `<BaseLayout ...>` and before `<div class="container">`.
